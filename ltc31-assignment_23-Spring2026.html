# A pixel is non-black if ANY of its 3 channels (R, G, B) is greater than 0
# np.any(..., axis=2) collapses the 3 colour channels into one True/False per pixel

mask = np.any(aus > 0, axis=2)

print(f"Sub-image shape: {aus.shape}")       # (height, width, 3)
print(f"Mask shape:      {mask.shape}")       # (height, width) — one value per pixel
print(f"Land pixels:     {mask.sum()}")
print(f"Black pixels:    {(~mask).sum()}")

# Collect all vegetation image files in sorted order (Jan → Dec)
image_files = sorted(image_dir.glob('2024??-nasa-vegetation.jpg'))

print(f"Found {len(image_files)} images:")
for f in image_files:
    print(f"  {f.name}")

# Row/column bounds for the Australia/Tasmania sub-image
ROW_START, ROW_END = 330, 540
COL_START, COL_END = 1150, 1380

# Store results
months     = []
avg_greens = []

for img_path in image_files:

    # Extract month number from filename (e.g. '202401-nasa-vegetation.jpg' → 1)
    month = int(img_path.stem[4:6])

    # Load image
    img = mpimg.imread(str(img_path))

    # Extract Australia/Tasmania sub-image
    aus = img[ROW_START:ROW_END, COL_START:COL_END]

    # Build boolean mask (non-black pixels = land)
    mask = np.any(aus > 0, axis=2)

    # Compute average green intensity for land pixels only
    avg_green = aus[:, :, 1][mask].mean()

    months.append(month)
    avg_greens.append(avg_green)

    print(f"Month {month:02d}: avg green = {avg_green:.2f}  (land pixels: {mask.sum()})")

# Plot
fig, ax = plt.subplots(figsize=(8, 4))

ax.plot(months, avg_greens, marker='o', linewidth=2, color='forestgreen')
ax.set_xlabel('Month', fontsize=12)
ax.set_ylabel('Average Green Intensity', fontsize=12)
ax.set_title('Monthly Average Greenness — Australia & Tasmania, 2024', fontsize=13)
ax.set_xticks(months)
ax.set_xticklabels(['Jan','Feb','Mar','Apr','May','Jun',
                    'Jul','Aug','Sep','Oct','Nov','Dec'])
ax.grid(True, linestyle='--', alpha=0.5)

plt.tight_layout()
plt.show()
